Class Detail
Tween()

A lightweight Tween class independant of Third Party libraries (aside from Robert Penner's easing functions). The engine has Paul Irish's requestAnimFrame shim built in allowing for consistent animations across browsers and devices. 

    
   Properties Example

   var tween = new Tween();
   target = document.getElementById("target");

   var propertyOptions = {
      node: target, 
      duration: 1000,
      properties:{height:{start:300, end:4000, unit:"px"},
                  width:{start:200, end:3000, unit:"px"}},
   }

   tween.to(propertyOptions);

   Curve Example

   var tween = new Tween();
   target = document.getElementById("target");

   var curveOptions = {
      node: target, 
      duration: 1000,
      curve:[0, 100],
      onAnimate:function(c){
       this.node.style.width = c + "px"
   }

   tween.to(curveOptions);

   Line Example

   var tween = new Tween();
   target = document.getElementById("target");

   var lineOptions = {
      node: target, 
      duration: 1000,
      curve:new Line([0, 0],[100,200]),
      onAnimate:function(c){
       this.node.style.width = c[0] + "px"
       this.node.style.left = c[1] + "px"
      }
   }

   tween.to(lineOptions);
