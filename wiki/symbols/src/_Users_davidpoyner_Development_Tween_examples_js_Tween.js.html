1 /** 
  2  *    @constructor 
  3  *    @description A lightweight <b>Tween</b> class independant of Third Party libraries (aside from Robert Penner's easing functions). The engine has Paul Irish's
  4  *    requestAnimFrame shim built in allowing for consistent animations across browsers and devices.
  5  *    <br>
  6  *    @example     
  7  *    <b><u>Properties Example</b></u>
  8  *
  9  *    var tween = new Tween();
 10  *    target = document.getElementById("target");
 11  *
 12  *    var propertyOptions = {
 13  *       node: target, 
 14  *       duration: 1000,
 15  *       properties:{height:{start:300, end:4000, unit:"px"},
 16  *                   width:{start:200, end:3000, unit:"px"}},
 17  *    }
 18  *
 19  *    tween.to(propertyOptions);
 20  *    @link <a href="../examples/PropertyExample.html">Property Example</a>
 21  *    @link <a href="../examples/CurveExample.html">Curve Example</a>
 22  *    @link <a href="../examples/LineExample.html">Line Example</a>
 23  *
 24  *    @example
 25  *
 26  *
 27  *    <b><u>Curve Example</b></u>
 28  *
 29  *    var tween = new Tween();
 30  *    target = document.getElementById("target");
 31  *
 32  *    var curveOptions = {
 33  *       node: target, 
 34  *       duration: 1000,
 35  *       curve:[0, 100],
 36  *       onAnimate:function(c){
 37  *        this.node.style.width = c + "px"
 38  *    }
 39  *
 40  *    tween.to(curveOptions);
 41  *    @example
 42  *
 43  *
 44  *    <b><u>Line Example</b></u>
 45  *
 46  *    var tween = new Tween();
 47  *    target = document.getElementById("target");
 48  *
 49  *    var lineOptions = {
 50  *       node: target, 
 51  *       duration: 1000,
 52  *       curve:new Line([0, 0],[100,200]),
 53  *       onAnimate:function(c){
 54  *        this.node.style.width = c[0] + "px"
 55  *        this.node.style.left = c[1] + "px"
 56  *       }
 57  *    }
 58  *
 59  *    tween.to(lineOptions);
 60  *    
 61  *
 62  *    @property {function} onEnd A function to be called at the end of the Tween.
 63  *    @property {function} onBegin A function to be called at the beginning of the Tween.
 64  *    @property {function} onAnimate A function to be called at every step of the Tween.
 65  *    @property {number} delay The delay before the  Tween starts.
 66  *    @property {boolean} isAnimating (read-only) Boolean determining if the Tween is in animating state.
 67  *    @property {boolean} isReversed (read-only) Boolean determining if the Tween is reversed.
 68  *    @property {boolean} isPaused (read-only) Boolean determining if the Tween is in the paused state.
 69  *    @property {boolean} isStopped (read-only) Boolean determining if the Tween is in the stopped state.
 70  *    @property {number} percentage (read-only) The percentage of the Tween's completion.
 71  *    @property {object} properties The properties object which is used to create the motion objects.
 72  *    @property {array} curve The curves object which is used to create the motion objects.
 73  *    @property {function} easing The easing function to be used for the tween.
 74  *    @property {object} node An object which the Tween will animate.
 75  *    @property {number} duration The duration of the tween (in milliseconds).
 76  *    @property {number} _previousTime (read-only) The previous time from the last step.
 77  *    @property {number} _currentTime (read-only) The current time for the current step.
 78  *    @property {number} _startTime (read-only) The initial start time for the tween.
 79  *    @property {number} _delta (read-only) The difference between the last and first step.
 80  *    @property {number} _t (read-only) The time value to pass to the easing function.
 81  *    @property {array}  _motionStack (read-only) The motion objects which will be tweened.
 82  */
 83 
 84 var Tween = function(){
 85   this.onEnd = -1; 
 86   this.onBegin = -1; 
 87   this.onAnimate = -1;  
 88   this.delay = 0;
 89   this.node = null;
 90   this.duration = 0;
 91   this.isAnimating = false; 
 92   this.isReversed = false; 
 93   this.isPaused = false; 
 94   this.isStopped = false;
 95   this.properties = -1;  
 96   this.curve = [0, 1];
 97   this.easing = Easing.Circ.easeIn;
 98   this._previousTime = null; 
 99   this._currentTime = null; 
100   this._startTime = null; 
101   this._delta = null; 
102   this._t = 0;
103   this._motionStack = [];
104 }

108    /**
109     * Private method which creates a MotionObject based on the curves set in the Tween Options before the start of the tween.
110     * @private  {object}    Tween._setCurve
111     * @property {object}
112     *
113    */
114    _setCurve:function(){
115     var c = this.curve;
116 
117     if (c instanceof Line == false){
118       var _m = new MotionObject();
119       _m.d = this.duration;
120 
121       this._motionStack.push(_m);
122     }else{
123         var _c1 = c.curves[0];
124         var _c2 = c.curves[1];
125 
126         for (var i = 0; i < _c1.length; ++i){
127           var _m = new MotionObject();
128 
129           _m.b = _c1[i];
130           _m.c = _c2[i] - _c1[i];
131           this._motionStack.push(_m);
132         }
133       }
134   },

136   /**
137    *  Private method which creates a MotionObject based on the property Object passed in
138    *  @private {object}     Tween._setProperty
139    *  @param   {object}     properties           The properties object which should contain standard CSS properties.
140    *  
141    */
142   
143    _setProperty:function(properties){
144          // Each object in the properties object should be a CSS style
145            for(property in properties){
146               var _m = new MotionObject();
147               _m.prop = property;
148               var _property = properties[property];
149               // If this is an object parse
150               // // TODO: webkit, color tranforms: further detection and proceeding parsing
151               //
152               if (typeof _property == "object"){
153                 for(_p in _property){
154                     if (_p == "begin") _m.b = _property[_p];
155                     if (_p == "end") {_m.c = _property[_p] - _m.b;}
156                     if (_p == "unit") _m.unit = _property[_p];
157                  }
158               // If not use the value as the end
159               }else{
160                  this.change = _property - this.begin;                 
161               }
162 
163               this._motionStack.push(_m);
164            }
165    },

167   /**
168    * Starts the tween according to the delay if any
169    * @private {object}    Tween._start
170    *
171    */
172 
173    _start:function(){
174     var self = this; // Self reference for the delay setTimeout callback
175 
176     this._currentTime = 0;
177    
178     if (this.onBegin != -1) this.onBegin(); 
179     this._previousTime = Date.now();
180     this.isAnimating = true;
181     this._t = (self._isReversed) ? self.duration : 0;
182 
183 
184     setTimeout(function(){
185         self._update();
186       }, self.delay); 
187    },
188 
189   /**
190    * Steps the tween 
191    * @private {object}    Tween._step
192    *
193    */
194 
195    _step:function(){
196         // Get the current time
197         this._currentTime = Date.now();
198         // Get the difference between the current time and the last time
199         this._delta = this._currentTime - this._previousTime;
200         // Bottleneck the difference if it is too high
201         this._delta = Math.min(this._delta, 25);
202         
203         // If we are moving forward
204         if (!this._isReversed){
205             // If the time and the difference is less than the duration
206             if (this._t + this._delta < this.duration){
207                 // Add this and the adjusted frame step to the tween value
208                 this._t = this._delta + this._t;
209                 // Continue to the next step
210                 this._update();
211             // If we are at the end of the tween
212             }else{
213                 // Set the tween value to the final step
214                 this._t = this.duration;
215                 // End the tween
216                 this._stop();
217             }
218         // If we are moving in reverse  
219         }else{
220              // If the tween value is greater than the start (0)
221              if (this._t > 0){
222                 // Tween value minus the adjusted frame step or the beginning value
223                 this._t = (this._t - this._delta > 0) ? this._t - this._delta : 0;
224                 // Continue to the next step
225                 this._update();
226             }else{
227               this._stop();
228             }
229           }
230 
231         // Iterate through the motion stack to get all our motion objects
232         for (tween in this._motionStack){
233             // Assign a temporary motion object
234             var motionObject = this._motionStack[tween];
235             // If it has a property value
236             if (motionObject.prop != -1){
237                 // Assign the value to the tween return value
238                 this.node.style[motionObject.prop] = this.easing( this._t, motionObject.b, motionObject.c, this.duration) + motionObject.unit;
239                 // If there is an onAnimate function return the tween with a beginning of 0 and an end of 1
240                 if (this.onAnimate != -1) var c = this.easing( this._t, 0, 1, this.duration);
241             // If there is no property value and only a curve value
242             }else{ 
243                 // If we only have one curve
244                 if(this._motionStack.length == 1){
245                     // Assign the onAnimate parameter to the one curve
246                     var c = this.easing( this._t, motionObject.b, motionObject.c, this.duration);
247                 // If there are multiple curves
248                 }else{
249                     // Assign the onAnimate parameter to an empty array
250                     var c = [];
251                     // Iterate through the motionObjects
252                     for (motionObject in this._motionStack){
253                         var _m = this._motionStack[motionObject];
254                         // Add the return paramater to the array
255                         c.push(this.easing( this._t, _m.b, _m.c, this.duration));
256                     }
257                 }
258              }
259           }
260 
261           // If there is an onAnimate callback
262           if (this.onAnimate != -1) this.onAnimate(c);
263           // Change the time
264           this._previousTime = this._currentTime;
265    },
266 
267   /**
268    * Stops the tween 
269    * @private {object}    Tween._stop
270    *
271    */
272 
273    _stop:function(){
274     this.isAnimating = false; this.isStopped = true;
275     if (this.onEnd != -1 && !this.isPaused) this.onEnd();
276    },

278   /**
279    * Updates the tween 
280    * @private {object}    Tween._update
281    *
282    */
283 
284    _update:function(c){
285     var self = this; // Self reference for the request animated frame callback
286     var hasOnAnimate = (this.onAnimate != -1)
287     if (this.isAnimating == true) requestAnimFrame(function(){self._step()});
288    },

290   /**
291    * Reverses the tween 
292    * @public {object}    Tween.reverse
293    *
294    */
295 
296    reverse:function(){
297      this.isReversed = (this.isReversed == false) ? this.isReversed = true : this.isReversed = false;
298    },

300   /**
301    * Checks to see if the tween is paused. If so, resumes the tween.
302    * @public {object}    Tween.resume
303    *
304    */
305 
306    resume:function(){
307      if (!this.isPaused) return;
308      // Reset our time settings
309      this._delta = 0;
310      this._previousTime = Date.now();
311      // Set our properties
312      this.isPaused = false;
313      this.isAnimating = true;
314      // Resume the tween
315      this._update();
316    },
317 
318   /**
319    * Pauses the tween 
320    * @public {object}    Tween.pause
321    *
322    */
323 
324    pause:function(){
325      // No need to pause a stopped tween
326      if (this.isStopped) return;
327      // Set the properties
328      this.isAnimating = false;
329      this.isPaused = true;
330      this._step();
331    },

333   /**
334    * Initializes the tween animation.
335    * @public {object}    Tween.to
336    * @param {object} options The options object.
337    * @example
338    * var options = {
339    *  node:this,
340    *  duration:100,
341    *  properties:{width:400}
342    * }
343    *
344    * tween.to(options)
345    */
346 
347    to:function(options){
348       //setup options
349       if(options){
350         // Iterate through the options
351         for (key in options){
352           // Assign our properties
353           this[key] = options[key];
354         }
355       }
356     
357       // Grab the motion objects
358       if (this.properties != -1) this._setProperty(this.properties);
359       else this._setCurve();
360 
361       // Let her rip
362       this._start();
363     }

365 /**
366   Create a MotionObject which the Tween Engine uses to modify a given value
367   @constructor
368   @property {number} b          the begin value for the tween
369   @property {number} c          the change value for the tween
370   @property {number} t          the time value for the tween
371   @property {string} prop       the property which the tween manipulates (if any)
372   @property {string} unit       the unit which determines the scale of the property (if any)
373 */
374 
375 function MotionObject(){
376   this.b = 0; this.c; this.t; this.prop = -1; this.unit;
377 }

379 /**
380   Create a Line Object or an array containing an array of start values and an array of end values
381   @constructor
382   @param {array} a              an Array of start points
383   @param {array} b              an Array of end points
384 */

386 function Line(a, b){
387   if (a.length != b.length) throw new Error("Uneven Amount of Lines " + a.length + " != " + b.length);
388   this.curves = [a, b];
389 }
